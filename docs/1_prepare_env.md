# Prepare environment


## Install packages

```
$ npm init
$ npm install webpack
$ npm install --save-dev css-loader style-loader
$ npm install --save-dev lodash
$ npm install --save-dev babel-core
$ npm install --save-dev babel-loader // Load js files
$ npm install --save-dev babel-preset-es2015 // ES6 to ES5
$ npm install --save-dev babel-polyfill // Symbol, Generator, Promise
$ npm install --save-dev babel-plugin-transform-class-properties // ES7 to ES5
$ npm install --save-dev babel-plugin-import // Antd import
$ npm install --save-dev babel-preset-react

$ npm install --save express
$ npm install --save antd
$ npm install --save ant-design-pro
$ npm install --save es6-promise
$ npm install --save express
$ npm install --save history
$ npm install --save isomorphic-fetch
$ npm install --save react
$ npm install --save react-dom
$ npm install --save react-redux
$ npm install --save react-responsive
$ npm install --save react-router
$ npm install --save react-router-dom
$ npm install --save react-router-redux
$ npm install --save redux
$ npm install --save redux-saga
$ npm install --save immutable // Record for reducer
```
